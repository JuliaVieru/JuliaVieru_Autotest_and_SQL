select c.login, count (o."inDelivery") as delivery_count from "Couriers" as c inner join "Orders" as o on c.id = o."courierId" when o."inDelivery" = 'true' group by c.login;


SELECT c.login, count (o."inDelivery") as delivery_count
FROM "Couriers" AS c
INNER JOIN "Orders" AS o ON c.id = o."courierId"
WHERE o."inDelivery" = 'true'
GROUP BY c.login;


2 task
select track, case when finished = true then '2' when cancelled = true then '-1' when "inDelivery" = true then '1' else '0' end as status from "Orders";


SELECT
    track,
    CASE
        WHEN finished = TRUE THEN 2
        WHEN cancelled = TRUE THEN -1
        WHEN "inDelivery" = TRUE THEN 1
        ELSE 0
    END AS status
FROM "Orders";
